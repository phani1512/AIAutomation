elem1 = driver.find_element(By.ID, "login-button")
# Scroll element into view
driver.execute_script("arguments[0].scrollIntoView({behavior: 'auto', block: 'center'});", elem1)

try:
    elem1.click()
except Exception as e:
    if 'intercepted' in str(e).lower() or 'not clickable' in str(e).lower():
        print('Element click intercepted, using JavaScript click')
        driver.execute_script('arguments[0].click();', elem1)
    else:
        raise
time.sleep(0.5)  # Brief pause after click

JavaScript/Playwright:

const elem1 = page.locator('#login-button');
// Scroll element into view
await elem1.scrollIntoViewIfNeeded();
await page.waitForTimeout(500);
// Try regular click, fallback to force click if intercepted
try {
    await elem1.click();
} catch (e) {
    if (e.message.includes('intercepted') || e.message.includes('not clickable')) {
        console.log('Element click intercepted, using JavaScript click');
        await elem1.evaluate(el => el.click());
    } else {
        throw e;
    }
}
await page.waitForTimeout(500);  // Brief pause after click
